% Layout of a subsection
\newenvironment{hsSubsection}[4]{ % 4 input arguments - company name, year(s) employed, job title and location
	{\bf #1} {\em (#3)} \hfill {#2} % Bold company name and date on the right
	\begin{list}{\tinybullet}{\leftmargin=1.5em\rightmargin=11em} % \cdot used for bullets, no indentation
		\itemsep -0.5em \vspace{-0.5em} % Compress items in list together for aesthetics
	}{

}

}
